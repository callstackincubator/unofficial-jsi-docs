116class JSI_EXPORT HostObject {
117 public:
118 // The C++ object's dtor will be called when the GC finalizes this
119 // object. (This may be as late as when the Runtime is shut down.)
120 // You have no control over which thread it is called on. This will
121 // be called from inside the GC, so it is unsafe to do any VM
122 // operations which require a Runtime&. Derived classes' dtors
123 // should also avoid doing anything expensive. Calling the dtor on
124 // a jsi object is explicitly ok. If you want to do JS operations,
125 // or any nontrivial work, you should add it to a work queue, and
126 // manage it externally.
127 virtual ~HostObject();
128
129 // When JS wants a property with a given name from the HostObject,
130 // it will call this method. If it throws an exception, the call
131 // will throw a JS \c Error object. By default this returns undefined.
132 // \return the value for the property.
133 virtual Value get(Runtime&, const PropNameID& name);
134
135 // When JS wants to set a property with a given name on the HostObject,
136 // it will call this method. If it throws an exception, the call will
137 // throw a JS \c Error object. By default this throws a type error exception
138 // mimicking the behavior of a frozen object in strict mode.
139 virtual void set(Runtime&, const PropNameID& name, const Value& value);
140
141 // When JS wants a list of property names for the HostObject, it will
142 // call this method. If it throws an exception, the call will throw a
143 // JS \c Error object. The default implementation returns empty vector.
144 virtual std::vector<PropNameID> getPropertyNames(Runtime& rt);
145};
